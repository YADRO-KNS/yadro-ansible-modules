Examples
--------

.. code-block:: yaml+jinja

    
    ---
    - name: Create TestUser
      yadro.obmc.bmc_account:
        connection:
          hostname: "localhost"
          username: "username"
          password: "password"
        username: "TestUser"
        password: "TestPassword"
        role: "Operator"
        enabled: true
        state: "present"

    - name: Modify TestUser
      yadro.obmc.bmc_account:
        connection:
          hostname: "localhost"
          username: "username"
          password: "password"
        username: "TestUser"
        enabled: false
        role: "ReadOnly"

    - name: Delete TestUser
      yadro.obmc.bmc_account:
        connection:
          hostname: "localhost"
          username: "username"
          password: "password"
        username: "TestUser"
        state: "absent"
